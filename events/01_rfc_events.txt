namespace = freecities

# Purchased charter
province_event = {
	id = freecities.1
	title = freecities.1.t
	desc = freecities.1.d
	picture = HRE_eventPicture
	goto = ROOT

	major = yes
	major_trigger = { is_emperor = yes }

	trigger = {
		NOT = { has_province_flag = releasable_free_city }
		owner = {
			is_core = ROOT
			capital_scope = { is_part_of_hre = yes }
			num_of_cities = 2
			OR = {
				is_bankrupt = yes
				is_in_large_debt = yes
			}
		}
		is_capital = no
		is_part_of_hre = yes
		development = 10
		#has_eligible_free_city_tag = yes
	}

	mean_time_to_happen = {
		months = 120

		modifier = {
			factor = 0.9
			development = 15
		}
		modifier = {
			factor = 0.9
			development = 20
		}
		modifier = {
			factor = 0.75
			development = 30
		}
		modifier = {
			factor = 0.9
			owner = { num_of_loans = 7 }
		}
		modifier = {
			factor = 0.9
			owner = { num_of_loans = 10 }
		}
		modifier = {
			factor = 0.5
			owner = { is_bankrupt = yes }
		}
	}

	immediate = {
		set_province_flag = releasable_free_city
	}

	option = {
		name = freecities.1.a
		add_development_scaled_income = yes
		owner = { add_prestige = -5 }
		custom_tooltip = releasable_free_city_tooltip
	}
}

# Charter won by dynasty extinction
country_event = {
	id = freecities.3
	title = none
	desc = none
	picture = none

	is_triggered_only = yes
	hidden = yes

	trigger = {
		can_have_same_dynasty_heir = yes
		NOT = { has_country_flag = succession_is_safe }
		capital_scope = { is_part_of_hre = yes }
		num_of_cities = 2
		any_owned_province = {
			NOT = { has_province_flag = releasable_free_city }
			is_core = ROOT
			is_capital = no
			is_part_of_hre = yes
			development = 10
			#has_eligible_free_city_tag = yes
		}
	}

	immediate = {
		random_owned_province = {
			limit = {
				NOT = { has_province_flag = releasable_free_city }
				is_core = ROOT
				is_capital = no
				is_part_of_hre = yes
				development = 10
				#has_eligible_free_city_tag = yes
			}
			add_province_modifier = {
				name = "free_city_charter_extinction"
				duration = 3650
			}
		}
	}

	option = {
	}
}

country_event = {
	id = freecities.7
	title = freecities.7.t
	desc = freecities.7.d
	picture = HRE_eventPicture
	goto = FROM

	is_triggered_only = yes

	immediate = {
		FROM = {
			add_province_modifier = {
				name = "offered_free_city_status"
				duration = 30
				hidden = yes
			}
			owner = { save_event_target_as = free_city_owner }
		}
	}

	option = {
		name = freecities.7.a
		release_free_city = yes
	}

	option = {
		name = freecities.7.b
		if = {
			limit = {
				NOT = {
					event_target:free_city_owner = {
						any_owned_province = {
							NOT = { has_province_modifier = offered_free_city_status }
							has_province_flag = releasable_free_city
							#has_eligible_free_city_tag = yes
						}
					}
				}
			}
			custom_tooltip = no_other_eligible_free_cities_tooltip
		}
	}

	after = {
		if = {
			limit = {
				OR = {
					AND = {
						hre_reform_passed = emperor_reichstag_collegia
						NOT = {
							calc_true_if = {
								all_country = {
									has_reform = free_city
								}
								amount = 15
							}
						}
					}
					NOT = {
						calc_true_if = {
							all_country = {
								has_reform = free_city
							}
							amount = 12
						}
					}
				}
				event_target:free_city_owner = {
					any_owned_province = {
						NOT = { has_province_modifier = offered_free_city_status }
						has_province_flag = releasable_free_city
						#has_eligible_free_city_tag = yes
					}
				}
			}
			event_target:free_city_owner = {
				random_owned_province = {
					limit = {
						NOT = { has_province_modifier = offered_free_city_status }
						has_province_flag = releasable_free_city
						#has_eligible_free_city_tag = yes
					}
					province_event = { id = freecities.6 }
				}
			}
		}
	}
}
